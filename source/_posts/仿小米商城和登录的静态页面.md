## login.js
```javascript
$(function () {
    $("#options li").click(function () {
        $(this).addClass("border").siblings().removeClass('border');
        var index = $(this).index();
        $(".main .item").eq(index).fadeToggle().show().siblings().hide()
    })

    $("#log_2").blur(function () {
        if ($("log_1").val() !== '' && $("log_2").val() !== '') {
            $("#log_3").addClass('sum')
        }
        else if ($("log_1").val() == '' || $("log_2").val() == '') {
            $("#log_3").removeClass('sum')
        }
    });
    $("#reg_2").blur(function () {
        if ($("reg_1").val() !== '' && $("reg_2").val() !== '') {
            $("#reg_3").addClass('sum')
        }
        else {
            $("#reg_3").removeClass('sum')
        }
    });

})
```
